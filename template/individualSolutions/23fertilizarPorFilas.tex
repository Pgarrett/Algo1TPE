\begin{problema}{fertilizarPorFilas}{s: Sistema}{}
\requiere[unDronPorFila]{sinRepetidos([prm(posicionActual(d)) | d \selec enjambreDrones(s), enVuelo(d)])}
\modifica{s}
\asegura[mismoCampoYCultivo]{mismoCampoYCutivo(s, pre(s))}
\asegura[mismaCantidadDeDrones]{|enjambreDrones(s)| == |enjambreDrones(pre(s))|}
\asegura[siNoVuelanNoSeModifican]{(\forall d' \selec enjambreDrones(pre(s))) \neg enVuelo(d') \Rightarrow droneEnSistema(s, d')}
